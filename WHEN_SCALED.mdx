# **Add Request ID Middleware** üÜî

**Create:** `packages/trpc/src/middleware/request-id.ts`

```typescript
import { v4 as uuidv4 } from "uuid";

export function addRequestId() {
  return async ({ ctx, next }: { ctx: any; next: any }) => {
    const requestId = uuidv4();
    return next({
      ctx: {
        ...ctx,
        requestId,
      },
    });
  };
}
```
