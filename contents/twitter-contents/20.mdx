<File label="style.css">

```css
.animate-text {
  opacity: 0;
  animation: reveal 0.5s var(--i) ease-in-out forwards;
}

@keyframes reveal {
  to {
    opacity: 1;
  }
}
```

</File>

<File label="app.js">

```js
const text = document.querySelector("h1");

text.innerHTML = text.textContent.replace(
  /\S/g,
  (char, index) =>
    `<span class='animate-text' style='--i:${index * 0.05}s'>${char}</span>`
);
```
