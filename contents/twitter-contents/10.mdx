tech: "js"

<Tabs>

<Tab label="Pause">

```js title="app.js"
const pauseButton = document.querySelector(".pause-button");
const animationElement = document.querySelector(".animate-progress");

pauseButton.addEventListener("click", () => {
  if (
    animationElement.style.animationPlayState === "running" ||
    animationElement.style.animationPlayState === ""
  ) {
    animationElement.style.animationPlayState = "paused";
    // ... existing code ...
  } else {
    animationElement.style.animationPlayState = "running";
    // ... existing code ...
  }
});
```

</Tab>

<Tab label="Restart">

```js title="app.js"
const pauseButton = document.querySelector(".pause-button");
const animationElement = document.querySelector(".animate-progress");

restartButton.addEventListener("click", () => {
  animationElement.classList.remove("animate-progress");

  requestAnimationFrame(() => {
    animationElement.classList.add("animate-progress");
  });
});
```

> Make sure that the `.animate-progress` is the class that the animation exists.

</Tab>
